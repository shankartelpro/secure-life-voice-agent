# --- TTS (ElevenLabs) ---
async def text_to_speech_stream(text: str):
    """
    Streams audio from ElevenLabs.
    Returns a generator that yields audio chunks.
    """
    log.info(f"TTS: Generating audio for: '{text[:30]}...'")
    
    url = f"https://api.elevenlabs.io/v1/text-to-speech/{settings.ELEVEN_VOICE_ID}/stream"
    
    headers = {
        "Accept": "audio/mpeg",
        "Content-Type": "application/json",
        "xi-api-key": settings.ELEVENLABS_API_KEY
    }
    
    # Model & Voice Settings for MAXIMUM NORMALCY
    data = {
        "text": text,
        "model_id": "eleven_turbo_v2_5",
        "voice_settings": {
            "stability": 0.4,         # Slightly higher than 0.3 for a more grounded/normal tone
            "similarity_boost": 1.0,  # MAX (1.0) - Sounds exactly like the original voice
            "clarity": 0.85           # High clarity for sharpness
        }
    }

    try:
        async with aiohttp.ClientSession() as session:
            async with session.post(url, json=data, headers=headers) as resp:
                if resp.status == 200:
                    log.info("TTS: Audio stream started successfully.")
                    async for chunk in resp.content.iter_chunked(16384): # Make sure this is 8192, not 1024
                        yield chunk
                else:
                    log.error(f"TTS Error: {resp.status} - {await resp.text()}")
                    yield b'' 
    except Exception as e:
        log.error(f"TTS Exception: {e}")
        yield b''

# --- STT (Deepgram) ---
async def deepgram_transcription_stream():
    """
    Connects to Deepgram WebSocket.
    Returns the WebSocket object (kept open).
    """
    uri = f"wss://api.deepgram.com/v1/listen?model={settings.DEEPGRAM_MODEL}&language={settings.DEEPGRAM_LANGUAGE}&smart_format=true"
    
    # NOTE: Changed 'extra_headers' to 'additional_headers' for websockets v14+
    extra_headers = {
        "Authorization": f"Token {settings.DEEPGRAM_API_KEY}"
    }

    log.info(f"Deepgram: Attempting connection to {uri}...")
    
    try:
        # FIXED: Use 'additional_headers' instead of 'extra_headers'
        ws = await websockets.connect(uri, additional_headers=extra_headers)
        log.info("Deepgram: Connection Established Successfully.")
        return ws
    except ConnectionRefusedError:
        log.error("Deepgram: Connection Refused. Check Internet or API URL.")
        raise
    except Exception as e:
        log.error(f"Deepgram: Failed to connect - {type(e).__name__}: {e}")
        raise e
